//===========================================================================
// Trigger: Unit Indexer
//
// This trigger works in two key phases:
// 1) During map initialization, enumerate all units of all players to give them an index.
// 2) Adds a second event to itself to index new units as they enter the map.
// As a unit enters the map, check for any old units that may have been removed at some point in order to free their index.
//===========================================================================
function Trig_Unit_Indexer_Func017Func004C takes nothing returns boolean
    if ( not ( udg_UDexRecycle == 0 ) ) then
        return false
    endif
    return true
endfunction

function Trig_Unit_Indexer_Func017C takes nothing returns boolean
    if ( not ( udg_UnitIndexerEnabled == true ) ) then
        return false
    endif
    return true
endfunction

function Trig_Unit_Indexer_Func030Func005C takes nothing returns boolean
    if ( not ( GetUnitUserData(udg_UDexUnits[udg_UDex]) == 0 ) ) then
        return false
    endif
    return true
endfunction

function Trig_Unit_Indexer_Func030C takes nothing returns boolean
    if ( not ( udg_UDexWasted == 15 ) ) then
        return false
    endif
    return true
endfunction

function Trig_Unit_Indexer_Func034C takes nothing returns boolean
    if ( not ( GetUnitUserData(GetFilterUnit()) == 0 ) ) then
        return false
    endif
    return true
endfunction

function Trig_Unit_Indexer_Actions takes nothing returns nothing
        call ExecuteFunc("InitializeUnitIndexer")
    endfunction
    //  
    // This is the most important function - it provides an index for units as they enter the map
    //  
    function IndexUnit takes nothing returns boolean
        local integer pdex= udg_UDex
    //  
    // You can use the boolean UnitIndexerEnabled to protect some of your undesirable units from being indexed
    // - Example:
    // -- Set UnitIndexerEnabled = False
    // -- Unit - Create 1 Dummy for (Triggering player) at TempLoc facing 0.00 degrees
    // -- Set UnitIndexerEnabled = True
    //  
    // You can also customize the following block - if conditions are false the (Matching unit) won't be indexed.
    //  
    if ( Trig_Unit_Indexer_Func017C() ) then
        //  
        // Generate a unique integer index for this unit
        //  
        if ( Trig_Unit_Indexer_Func017Func004C() ) then
            set udg_UDex=( udg_UDexGen + 1 )
            set udg_UDexGen=udg_UDex
        else
            set udg_UDex=udg_UDexRecycle
            set udg_UDexRecycle=udg_UDexNext[udg_UDex]
        endif
        //  
        // Link index to unit, unit to index
        //  
        set udg_UDexUnits[udg_UDex]=GetFilterUnit()
        call SetUnitUserData(udg_UDexUnits[udg_UDex], udg_UDex)
        //  
        // Use a doubly-linked list to store all active indexes
        //  
        set udg_UDexPrev[udg_UDexNext[0]]=udg_UDex
        set udg_UDexNext[udg_UDex]=udg_UDexNext[0]
        set udg_UDexNext[0]=udg_UDex
        //  
        // Fire index event for UDex
        //  
        set udg_UnitIndexEvent=0.00
        set udg_UnitIndexEvent=1.00
        set udg_UnitIndexEvent=0.00
        set udg_UDex=pdex
    else
    endif
        return false
    endfunction
    //  
    // The next function is called each time a unit enters the map
    //  
    function IndexNewUnit takes nothing returns boolean
        local integer pdex= udg_UDex
        local integer ndex
    //  
    // Recycle indices of units no longer in-play every (15) units created
    //  
    set udg_UDexWasted=( udg_UDexWasted + 1 )
    if ( Trig_Unit_Indexer_Func030C() ) then
        set udg_UDexWasted=0
        set udg_UDex=udg_UDexNext[0]
        loop
            exitwhen udg_UDex == 0
        if ( Trig_Unit_Indexer_Func030Func005C() ) then
            //  
            // Remove index from linked list
            //  
            set ndex=udg_UDexNext[udg_UDex]
            set udg_UDexNext[udg_UDexPrev[udg_UDex]]=ndex
            set udg_UDexPrev[ndex]=udg_UDexPrev[udg_UDex]
            set udg_UDexPrev[udg_UDex]=0
            //  
            // Fire deindex event for UDex
            //  
            set udg_UnitIndexEvent=2.00
            set udg_UnitIndexEvent=0.00
            //  
            // Recycle the index for later use
            //  
            set udg_UDexUnits[udg_UDex]=null
            set udg_UDexNext[udg_UDex]=udg_UDexRecycle
            set udg_UDexRecycle=udg_UDex
            set udg_UDex=ndex
        else
            set udg_UDex=udg_UDexNext[udg_UDex]
        endif
        endloop
        set udg_UDex=pdex
    else
    endif
    //  
    // Handle the entering unit (Matching unit)
    //  
    if ( Trig_Unit_Indexer_Func034C() ) then
        call IndexUnit()
    else
    endif
        return false
    endfunction
    //  
    // The next function initializes the core of the system
    //

function Trig_Meathook_loop_Func001A takes nothing returns nothing
    set udg_MeatHHead=GetEnumUnit()
    set udg_MeatHRef=GetUnitUserData(udg_MeatHHead)
    set udg_MeatHCX=GetUnitX(udg_MeatHC[udg_MeatHRef])
    set udg_MeatHCY=GetUnitY(udg_MeatHC[udg_MeatHRef])
    // ---------------------------------------------
    // Index row group into array
    // ---------------------------------------------
    call ForGroupBJ(udg_MeatHRow[udg_MeatHRef], function Trig_Meathook_loop_Func001Func008A)
    // ---------------------------------------------
    // Move the chain
    // ---------------------------------------------
    if ( Trig_Meathook_loop_Func001Func012C() ) then
        if ( Trig_Meathook_loop_Func001Func012Func001C() ) then
            if ( Trig_Meathook_loop_Func001Func012Func001Func001C() ) then
                set udg_MeatHDeg[udg_MeatHRef]=GetUnitFacing(udg_MeatHC[udg_MeatHRef])
            else
                set udg_tempX=GetUnitX(udg_MeatHLockOn[udg_MeatHRef])
                set udg_tempY=GetUnitY(udg_MeatHLockOn[udg_MeatHRef])
                set udg_MeatHDeg[udg_MeatHRef]=bj_RADTODEG * Atan2(udg_tempY - udg_MeatHChainY[udg_MeatHRef], udg_tempX - udg_MeatHChainX[udg_MeatHRef])
            endif
        else
        endif
        set udg_tempX=( udg_MeatHChainX[udg_MeatHRef] + ( udg_MeatHSpeed[udg_MeatHRef] * CosBJ(udg_MeatHDeg[udg_MeatHRef]) ) )
        set udg_tempY=( udg_MeatHChainY[udg_MeatHRef] + ( udg_MeatHSpeed[udg_MeatHRef] * SinBJ(udg_MeatHDeg[udg_MeatHRef]) ) )
        // ---------------------------------------------
        // If hook hit map boundary, return the hook. Else move it forward
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func012Func007C() ) then
            set udg_UDex=udg_MeatHRef
            set udg_MeatHEvent=5.00
            set udg_MeatHEvent=0.00
            set udg_MeatHLaunch[udg_MeatHRef]=false
        else
            call SetUnitX(udg_MeatHHead, udg_tempX)
            call SetUnitY(udg_MeatHHead, udg_tempY)
            set udg_MeatHChainX[udg_MeatHRef]=udg_tempX
            set udg_MeatHChainY[udg_MeatHRef]=udg_tempY
        endif
    else
    endif
    if ( Trig_Meathook_loop_Func001Func013C() ) then
        // ---------------------------------------------
        // If Launch = true, move the chain forward
        // ---------------------------------------------
        set udg_Temp_IntB=2
        loop
            exitwhen udg_Temp_IntB > udg_MeatHSize[udg_MeatHRef]
            // move chain towards head
            set udg_Temp_IntC=( udg_Temp_IntB - 1 )
            set udg_Temp_Real=bj_RADTODEG * Atan2(udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]] - udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntC]], udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]] - udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntC]])
            set udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]]=( udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntC]] + ( udg_MeatHChainRate[udg_MeatHRef] * CosBJ(udg_Temp_Real) ) )
            set udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]]=( udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntC]] + ( udg_MeatHChainRate[udg_MeatHRef] * SinBJ(udg_Temp_Real) ) )
            call SetUnitX(udg_MeatHChain[udg_Temp_IntB], udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]])
            call SetUnitY(udg_MeatHChain[udg_Temp_IntB], udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]])
            call SetUnitFacing(udg_MeatHChain[udg_Temp_IntB], udg_Temp_Real)
            set udg_Temp_IntB=udg_Temp_IntB + 1
        endloop
        set udg_MeatHTraveled[udg_MeatHRef]=( udg_MeatHTraveled[udg_MeatHRef] + udg_MeatHSpeed[udg_MeatHRef] )
        // Set Launce to fasle when the maximum distance is reached
        if ( Trig_Meathook_loop_Func001Func013Func011C() ) then
            set udg_UDex=udg_MeatHRef
            set udg_MeatHEvent=5.00
            set udg_MeatHEvent=0.00
            set udg_MeatHLaunch[udg_MeatHRef]=false
        else
        endif
    else
        // ---------------------------------------------
        // If Launch = false, check if grapple is true
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func013Func007C() ) then
            // ---------------------------------------------
            // If grapple is true and a target is in the hook...
            // ---------------------------------------------
            if ( Trig_Meathook_loop_Func001Func013Func007Func007C() ) then
                // ---------------------------------------------
                // If target is alive, move grapple head to target
                // ---------------------------------------------
                set udg_MeatHNewPoint=GetUnitLoc(udg_MeatHT[udg_MeatHRef])
                set udg_MeatHChainX[udg_MeatHRef]=GetLocationX(udg_MeatHNewPoint)
                set udg_MeatHChainY[udg_MeatHRef]=GetLocationY(udg_MeatHNewPoint)
                call RemoveLocation(udg_MeatHNewPoint)
                call SetUnitX(udg_MeatHHead, udg_MeatHChainX[udg_MeatHRef])
                call SetUnitY(udg_MeatHHead, udg_MeatHChainY[udg_MeatHRef])
                set udg_Temp_IntB=2
                loop
                    exitwhen udg_Temp_IntB > udg_MeatHSize[udg_MeatHRef]
                    set udg_Temp_Real=bj_RADTODEG * Atan2(udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]] - udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB - 1]], udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]] - udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB - 1]])
                    set udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]]=( udg_MeatHChainX[udg_MeatHChainRef[( udg_Temp_IntB - 1 )]] + ( udg_MeatHChainRate[udg_MeatHRef] * CosBJ(udg_Temp_Real) ) )
                    set udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]]=( udg_MeatHChainY[udg_MeatHChainRef[( udg_Temp_IntB - 1 )]] + ( udg_MeatHChainRate[udg_MeatHRef] * SinBJ(udg_Temp_Real) ) )
                    call SetUnitX(udg_MeatHChain[udg_Temp_IntB], udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]])
                    call SetUnitY(udg_MeatHChain[udg_Temp_IntB], udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]])
                    call SetUnitFacing(udg_MeatHChain[udg_Temp_IntB], udg_Temp_Real)
                    set udg_Temp_IntB=udg_Temp_IntB + 1
                endloop
            else
                set udg_MeatHT[udg_MeatHRef]=null
            endif
            if ( Trig_Meathook_loop_Func001Func013Func007Func008C() ) then
                // ---------------------------------------------
                // If the caster is not hooked by someone else, move caster along the chain
                // ---------------------------------------------
                set udg_Temp_Real=bj_RADTODEG * Atan2(udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] - udg_MeatHCY, udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] - udg_MeatHCX)
                set udg_MeatHCX=( udg_MeatHCX + ( udg_MeatHSpeed[udg_MeatHRef] * CosBJ(udg_Temp_Real) ) )
                set udg_MeatHCY=( udg_MeatHCY + ( udg_MeatHSpeed[udg_MeatHRef] * SinBJ(udg_Temp_Real) ) )
                call SetUnitX(udg_MeatHC[udg_MeatHRef], udg_MeatHCX)
                call SetUnitY(udg_MeatHC[udg_MeatHRef], udg_MeatHCY)
            else
            endif
        else
            // ---------------------------------------------
            // If grapple is false, return the chain to the caster
            // ---------------------------------------------
            set udg_Temp_IntB=1
            loop
                exitwhen udg_Temp_IntB > ( udg_MeatHSize[udg_MeatHRef] - 1 )
                // move chain, starting at the head
                set udg_Temp_Real=bj_RADTODEG * Atan2(udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB + 1]] - udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]], udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB + 1]] - udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]])
                set udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]]=( udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]] + ( udg_MeatHSpeed[udg_MeatHRef] * CosBJ(udg_Temp_Real) ) )
                set udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]]=( udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]] + ( udg_MeatHSpeed[udg_MeatHRef] * SinBJ(udg_Temp_Real) ) )
                call SetUnitX(udg_MeatHChain[udg_Temp_IntB], udg_MeatHChainX[udg_MeatHChainRef[udg_Temp_IntB]])
                call SetUnitY(udg_MeatHChain[udg_Temp_IntB], udg_MeatHChainY[udg_MeatHChainRef[udg_Temp_IntB]])
                call SetUnitFacing(udg_MeatHChain[udg_Temp_IntB], udg_Temp_Real)
                set udg_Temp_IntB=udg_Temp_IntB + 1
            endloop
            set udg_Temp_Real=bj_RADTODEG * Atan2(udg_MeatHCY - udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]], udg_MeatHCX - udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]])
            set udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]]=( udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] + ( udg_MeatHSpeed[udg_MeatHRef] * CosBJ(udg_Temp_Real) ) )
            set udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]]=( udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] + ( udg_MeatHSpeed[udg_MeatHRef] * SinBJ(udg_Temp_Real) ) )
            call SetUnitX(udg_MeatHChain[udg_MeatHSize[udg_MeatHRef]], udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]])
            call SetUnitY(udg_MeatHChain[udg_MeatHSize[udg_MeatHRef]], udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]])
            if ( Trig_Meathook_loop_Func001Func013Func007Func015C() ) then
                // ---------------------------------------------
                // Move target to the head
                // ---------------------------------------------
                call SetUnitX(udg_MeatHT[udg_MeatHRef], udg_MeatHChainX[udg_MeatHRef])
                call SetUnitY(udg_MeatHT[udg_MeatHRef], udg_MeatHChainY[udg_MeatHRef])
            else
            endif
        endif
    endif
    // ---------------------------------------------
    // Add / Remove chains
    // ---------------------------------------------
    set udg_Temp_Real=distance(udg_MeatHCX , udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] , udg_MeatHCY , udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]]) - udg_MeatHOffset[udg_MeatHRef]
    if ( (udg_MeatHLaunch[udg_MeatHRef] == false) ) then // INLINED!!
        set udg_Temp_Real=( udg_Temp_Real - udg_MeatHSpeed[udg_MeatHRef] )
    else
        // ...
    endif
    if ( Trig_Meathook_loop_Func001Func019C() ) then
        if ( Trig_Meathook_loop_Func001Func019Func001C() ) then
            loop
            exitwhen udg_Temp_Real < udg_MeatHChainRate[udg_MeatHRef]
            set udg_MeatHAngle=bj_RADTODEG * Atan2(udg_MeatHCY - udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]], udg_MeatHCX - udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]])
            set udg_MeatHTempX=( udg_MeatHChainX[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] + ( udg_MeatHChainRate[udg_MeatHRef] * CosBJ(udg_MeatHAngle) ) )
            set udg_MeatHTempY=( udg_MeatHChainY[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]] + ( udg_MeatHChainRate[udg_MeatHRef] * SinBJ(udg_MeatHAngle) ) )
            set udg_Temp_Int=GetUnitUserData(CreateUnit(Player(PLAYER_NEUTRAL_PASSIVE), udg_dummy, udg_MeatHTempX, udg_MeatHTempY, udg_MeatHAngle))
            call PauseUnitBJ(true, udg_UDexUnits[udg_Temp_Int])
            call GroupAddUnitSimple(udg_UDexUnits[udg_Temp_Int], udg_MeatHRow[udg_MeatHRef])
            set udg_MeatHSize[udg_MeatHRef]=( udg_MeatHSize[udg_MeatHRef] + 1 )
            set udg_MeatHLst[udg_Temp_Int]=udg_MeatHSize[udg_MeatHRef]
            set udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]=udg_Temp_Int
            set udg_MeatHChainX[udg_Temp_Int]=udg_MeatHTempX
            set udg_MeatHChainY[udg_Temp_Int]=udg_MeatHTempY
            call SetUnitScalePercent(udg_UDexUnits[udg_Temp_Int], udg_MeatHChainScale[udg_MeatHRef], udg_MeatHChainScale[udg_MeatHRef], udg_MeatHChainScale[udg_MeatHRef])
            call AddSpecialEffectTargetUnitBJ("orign", udg_UDexUnits[udg_Temp_Int], udg_MeatHChainModel[udg_MeatHRef])
            set udg_MeatHEffect[udg_Temp_Int]=GetLastCreatedEffectBJ()
            set udg_MeatHChain[udg_MeatHSize[udg_MeatHRef]]=udg_UDexUnits[udg_Temp_Int]
            set udg_MeatHChainRate[udg_MeatHRef]=( udg_MeatHChainRate[udg_MeatHRef] + 0.10 )
            set udg_Temp_Real=( udg_Temp_Real - udg_MeatHChainRate[udg_MeatHRef] )
            endloop
        else
        endif
    else
    endif
    if ( Trig_Meathook_loop_Func001Func020C() ) then
        loop
        exitwhen udg_Temp_Real > udg_MeatHSpeed[udg_MeatHRef]
        // ---------------------------------------------
        // If only the head is left and is returning to the caster, end spell
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func020Func006C() ) then
            set udg_UDex=udg_MeatHRef
            set udg_MeatHEvent=3.00
            set udg_MeatHEvent=0.00
            call TriggerExecute(gg_trg_Meathook_end)
            set udg_Temp_Real=( udg_MeatHSpeed[udg_MeatHRef] + 1 )
        else
            if ( Trig_Meathook_loop_Func001Func020Func006Func005C() ) then
                call DestroyEffectBJ(udg_MeatHEffect[udg_MeatHChainRef[udg_MeatHSize[udg_MeatHRef]]])
                call RemoveUnit(udg_MeatHChain[udg_MeatHSize[udg_MeatHRef]])
                set udg_MeatHSize[udg_MeatHRef]=( udg_MeatHSize[udg_MeatHRef] - 1 )
                set udg_Temp_Real=( udg_Temp_Real + udg_MeatHChainRate[udg_MeatHRef] )
                set udg_MeatHChainRate[udg_MeatHRef]=( udg_MeatHChainRate[udg_MeatHRef] - 0.10 )
            else
            endif
        endif
        endloop
    else
    endif
    set udg_UDex=udg_MeatHRef
    set udg_MeatHEvent=1.00
    set udg_MeatHEvent=0.00
    // ---------------------------------------------
    // if spell still exists...
    // ---------------------------------------------
    if ( Trig_Meathook_loop_Func001Func027C() ) then
        // ---------------------------------------------
        // Search for a target to hook
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func027Func004C() ) then
            if ( Trig_Meathook_loop_Func001Func027Func004Func001C() ) then
                call GroupEnumUnitsInRange(udg_Temp_Targets, udg_MeatHChainX[udg_MeatHRef], udg_MeatHChainY[udg_MeatHRef], udg_MeatHAoE[udg_MeatHRef], Condition(function FilterLockedUnit))
            else
                call GroupEnumUnitsInRange(udg_Temp_Targets, udg_MeatHChainX[udg_MeatHRef], udg_MeatHChainY[udg_MeatHRef], udg_MeatHAoE[udg_MeatHRef], null)
            endif
            call ForGroupBJ(udg_Temp_Targets, function Trig_Meathook_loop_Func001Func027Func004Func002A)
            call GroupClear(udg_Temp_Targets)
        else
        endif
        // ---------------------------------------------
        // Deal damage
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func027Func008C() ) then
            call UnitDamageTargetBJ(udg_MeatHC[udg_MeatHRef], udg_MeatHT[udg_MeatHRef], udg_MeatHDamageLoop[udg_MeatHRef], udg_MeatHDamageType[udg_MeatHRef], DAMAGE_TYPE_NORMAL)
            call DestroyEffect(AddSpecialEffectTarget(udg_MeatHSFX[udg_MeatHRef], udg_MeatHT[udg_MeatHRef], "chest"))
            if ( Trig_Meathook_loop_Func001Func027Func008Func003C() ) then
                set udg_UDex=udg_MeatHRef
                set udg_MeatHEvent=4.00
                set udg_MeatHEvent=0.00
                set udg_MeatHT[udg_MeatHRef]=null
                if ( (udg_MeatHRecicle[udg_MeatHRef] == true) ) then // INLINED!!
                    set udg_MeatHFull[udg_MeatHRef]=false
                else
                    // ...
                endif
            else
            endif
        else
        endif
        // ---------------------------------------------
        // Deal AoE damage
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func027Func012C() ) then
            call GroupEnumUnitsInRange(udg_Temp_Targets, udg_MeatHChainX[udg_MeatHRef], udg_MeatHChainY[udg_MeatHRef], udg_MeatHDamageAoE[udg_MeatHRef], null)
            call ForGroupBJ(udg_Temp_Targets, function Trig_Meathook_loop_Func001Func027Func012Func002A)
        else
        endif
        // ---------------------------------------------
        // Destroy destructables
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func027Func016C() ) then
            set udg_Temp_Rect=Rect(( udg_MeatHChainX[udg_MeatHRef] - udg_MeatHDamageAoE[udg_MeatHRef] ), ( udg_MeatHChainY[udg_MeatHRef] - udg_MeatHDamageAoE[udg_MeatHRef] ), ( udg_MeatHChainX[udg_MeatHRef] + udg_MeatHDamageAoE[udg_MeatHRef] ), ( udg_MeatHChainY[udg_MeatHRef] + udg_MeatHDamageAoE[udg_MeatHRef] ))
            call EnumDestructablesInRectAll(udg_Temp_Rect, function Trig_Meathook_loop_Func001Func027Func016Func002A)
            call RemoveRect(udg_Temp_Rect)
        else
        endif
        // ---------------------------------------------
        // If Locked on target is dead, remove target
        // ---------------------------------------------
        if ( Trig_Meathook_loop_Func001Func027Func020C() ) then
            set udg_MeatHLockOn[udg_MeatHRef]=null
        else
        endif
    else
    endif
endfunction

function Trig_Meathook_loop_Actions takes nothing returns nothing
    call ForGroupBJ(udg_MeatHHeads, function Trig_Meathook_loop_Func001A)
endfunction

    set udg_Temp_Int=GetUnitUserData(GetEnumUnit())
    call DestroyEffectBJ(udg_MeatHEffect[udg_Temp_Int])
    call KillUnit(udg_UDexUnits[udg_Temp_Int])
endfunction

function Trig_Meathook_end_Func001C takes nothing returns boolean
    if ( not ( IsUnitInGroup(udg_UDexUnits[udg_UDex], udg_MeatHHeads) == true ) ) then
        return false
    endif
    return true
endfunction

function Trig_Meathook_end_Actions takes nothing returns nothing
    if ( Trig_Meathook_end_Func001C() ) then
        call SetUnitPathing(udg_MeatHC[udg_UDex], true)
        call SetUnitPathing(udg_MeatHT[udg_UDex], true)
        set udg_MeatHHooked[GetUnitUserData(udg_MeatHT[udg_UDex])]=0
        set udg_MeatHGrapling[GetUnitUserData(udg_MeatHC[udg_UDex])]=0
        set udg_MeatHSpell[udg_UDex]=0
        set udg_MeatHT[udg_UDex]=null
        set udg_MeatHLockOn[udg_UDex]=null
        set udg_MeatHGrapple[udg_UDex]=false
        set udg_MeatHRecicle[udg_UDex]=false
        set udg_MeatHDestroyTrees[udg_UDex]=false
        set udg_MeatHOffset[udg_UDex]=0.00
        set udg_MeatHChainScale[udg_UDex]=0.00
        set udg_MeatHDist[udg_UDex]=0.00
        set udg_MeatHTraveled[udg_UDex]=0.00
        set udg_MeatHSpeed[udg_UDex]=0.00
        set udg_MeatHDamage[udg_UDex]=0.00
        set udg_MeatHDamageLoop[udg_UDex]=0.00
        set udg_MeatHAoEDamage[udg_UDex]=0.00
        set udg_MeatHAoE[udg_UDex]=0.00
        set udg_MeatHChainRate[udg_UDex]=0.00
        set udg_MeatHDamageType[udg_UDex]=null
        set udg_MeatHChainModel[udg_UDex]=null
        set udg_MeatHAngle=0.00
        call ForGroupBJ(udg_MeatHRow[udg_UDex], function Trig_Meathook_end_Func001Func025A)
        call GroupRemoveUnitSimple(udg_UDexUnits[udg_UDex], udg_MeatHHeads)
        call DestroyGroup(udg_MeatHRow[udg_UDex])
        set udg_MeatHCount=( udg_MeatHCount - 1 )
        if udg_MeatHCount == 0 then
        call DisableTrigger(gg_trg_Meathook_loop)
        endif
    else
        call DisplayTextToForce(GetPlayersAll(), ( "ERROR: Attempted to remove an invalid meathook with ID " + I2S(udg_UDex) ))
    endif
endfunction

    call DisableTrigger(gg_trg_Meathook_end)
    call TriggerAddAction(gg_trg_Meathook_end, function Trig_Meathook_end_Actions)
endfunction

function Trig_Meathook_Conditions takes nothing returns boolean

function Trig_Meathook_Func027C takes nothing returns boolean
    if ( not ( udg_Config_MeatHLockOn == null ) ) then
        return false
    endif
    return true
endfunction

function Trig_Meathook_Actions takes nothing returns nothing
    set udg_Config_MeatHSpell='A002'
    set udg_Config_MeatHlvl=GetUnitAbilityLevelSwapped('A002', GetTriggerUnit())
    // set the angle
    set udg_Temp_Point=GetUnitLoc(GetTriggerUnit())
    set udg_Temp_Point2=GetSpellTargetLoc()
    set udg_Config_MeatHAngle=AngleBetweenPoints(udg_Temp_Point, udg_Temp_Point2)
    call RemoveLocation(udg_Temp_Point)
    call RemoveLocation(udg_Temp_Point2)
    // -------------
    // If Config_MeatHHook is false then this variables means nothing (you can't grapple something if your hook doesn't accepts a target)
    set udg_Config_MeatHGrapple=false
    // if the target die in a hook, can the same hook hook a new target?
    set udg_Config_MeatHRecicle=true
